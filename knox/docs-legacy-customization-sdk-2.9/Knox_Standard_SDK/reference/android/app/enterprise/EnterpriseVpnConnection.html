 // Example code of how to get CommonName from certificate:
 X509Certificate cert = loadCertificate(); // example method to
 // read certificate
 String certCn = null;
 X500Principal principal = cert.getSubjectX500Principal();
 certNameArray = principal.getName().split(",");
 for (String field : certNameArray) {
     if (field.contains("CN=")) {
         certCn = field.replace("CN=", "");
     }
 }
 if (certCn == null) {
     certCn = certNameArray[0].substring(0, certNameArray[0].indexOf("=#"));
 }

public byte[] certHash

Since: API Level MDM_2.0

This field is deprecated.
Since MDM 5.7

Fingerprint of the certificate used in the connection. This should be the SHA-1 hash of the certificate to be used in bytes. 

 // Example code of how to get SHA-1 hash from certificate:
 try {
     EnterpriseVpnConnection conn = new EnterpriseVpnConnection();
     conn.type = EnterpriseVpnPolicy.VPN_TYPE_ANYCONNECT;
     X509Certificate cert = loadCertificate(); // example method to
                                               // read certificate
     MessageDigest md = MessageDigest.getInstance("SHA-1");
     byte[] der = cert.getEncoded();
     md.update(der);
     byte[] certHash = md.digest();
 } catch (NoSuchAlgorithmException e) {
     e.printStackTrace();
 } catch (CertificateException e) {
     e.printStackTrace();
 }
